        const res = await axios.post(`http://localhost:3000/api/v1/user/logout`,{},{
          headers: {
            Authorization: `Bearer ${accessToken}`
          }
          
        });

          toast.success('Logged out successfully');
        }
        console.log("TOKEN:", accessToken);
      } catch (error) {
        console.log(error);
        ;
      }
    };
